\subsection{hs015 test problem}

hs015.mod
\begin{equation*}
\begin{array}{lll}
\underset{x}{\text{minimize}} 	& 100(x_2 - x_1^2)^2 + (1 - x_1)^2 	& \\
\text{subject to} 				& 1 \le x_1 x_2 					& \\
								& 0 \le x_1 + x_2^2 				& \\
								& x_1 \le \frac{1}{2} 				& \\
\end{array}
\end{equation*}

initial primal $x^{(0)} = (-2, 1)$ and dual $\lambda^{(0)} = (0, 0)$.
